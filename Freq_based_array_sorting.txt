//Program to print array in frequency based sorted order

#include <bits/stdc++.h>
using namespace std;

void freqsort(vector<int>&arr)
{
    
    unordered_map<int,int>freq;
    
    for(int x:arr)
        freq[x]++;
        
    auto  cmp=[&](int a, int b)
    {
        if(freq[a]!= freq[b]) 
           return freq[a]>freq[b];
        return a<b;
    };
    
    sort(arr.begin(),arr.end(),cmp);
}

int main()
{
    vector<int>arr;
    int n;
    
    while(cin>>n)
       arr.push_back(n);
       
    freqsort(arr);
    
    for(int x:arr)
       cout<<x<<" ";
    
    return 0;
}
