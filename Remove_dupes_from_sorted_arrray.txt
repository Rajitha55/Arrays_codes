#include <bits/stdc++.h>
using namespace std;

//To print only distinct elements from a sorted array

int removedupes(vector<int>&a)
{
    int i=0;
    for(int j=1;j<a.size();j++)
    {
        if(a[i]!=a[j])
        {
            i++;
            a[i]=a[j];
        }
    }
    return i;
}

int main()
{
    vector<int>a;
    int x;
    //We are taking elements in sorted order for the array
    cout<<"Enter elements for Array: ";
    while(cin>>x)
      a.push_back(x);
    
    int k=removedupes(a);
    
    for(int i=0;i<=k;i++)
       cout<<a[i];
       
    return 0;
}
