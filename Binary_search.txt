//Search an element in an array and return its position
//If element not found, then return -1
//It works only when the array is in sorted order

#include <bits/stdc++.h>
using namespace std;

int search(vector<int>arr,int n)
{
    int low=0,high=arr.size()-1;
    int mid;
    while(low<=high)
    {
        mid=(low+high)/2;
        if(arr[mid]==n)
           return mid;
        else if(n<arr[mid])
           high=mid-1;
        else if(n>arr[mid])
           low=mid+1;
    }
    
    return -1;
}

int main()
{
    int n,ele;
    cout<<"Enter no.of elements in an array: ";
    cin>>n;
    vector<int>arr(n);
    cout<<"Enter elements for an array:"<<endl;
    for(int i=0;i<n;i++)
        cin>>arr[i];
       
    cout<<"Enter element to search in an array: ";
    cin>>ele;
    
    int res=search(arr,ele);
    cout<<"Element "<<ele<<" found at the index: "<<res;
    
    return 0;
}
