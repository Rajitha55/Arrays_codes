//Program to check whether the given array arr1 is subset of other given array arr2.
//If all the elements of arr1 are present in arr2, then we can say that arr1 is subset of arr2.

#include <bits/stdc++.h>
using namespace std;

bool search(vector<int>arr2, int ele)
{
    int low=0,high=arr2.size()-1;
    while(low<=high)
    {
        int mid=(low+high)/2;
        if(ele==arr2[mid])
           return true;
        else if(ele<arr2[mid]) 
           high=mid-1;
        else 
           low=mid+1;
    }
    return false;
}


bool subarr(vector<int>arr1, vector<int>arr2)
{
    if(arr1.size()>arr2.size())
       return false;
    sort(arr2.begin(), arr2.end());
    
    for(int x:arr1)
    {
        bool found=search(arr2,x);
        if(found==false)
            return false;
    }
    return true;
}

int main()
{
    int m,n;
    cout<<"Enter no.of elements in arr1: ";
    cin>>m;
    vector<int>arr1(m);
    cout<<"Enter elements for arr1:"<<endl;
    for(int i=0;i<m;i++)
       cin>>arr1[i];
     
    cout<<"Enter no.of elements for arr2:" ;
    cin>>n;
    vector<int>arr2(n);   
    cout<<"Enter elements for arr2:"<<endl;
    for(int i=0;i<n;i++)
       cin>>arr2[i];
    
    bool res=subarr(arr1,arr2);
    
    if(res==true) 
       cout<<"arr1 is subset of arr2";
    else
       cout<<"arr1 is not a subset of arr2";
       
    return 0;
}
